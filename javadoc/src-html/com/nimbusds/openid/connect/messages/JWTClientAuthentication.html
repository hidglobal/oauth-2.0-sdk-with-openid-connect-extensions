005    import java.util.Map;
006    
007    import javax.mail.internet.ContentType;
008    
009    import net.minidev.json.JSONObject;
010    
011    import com.nimbusds.jwt.JWA;
012    import com.nimbusds.jwt.JWTException;
013    import com.nimbusds.jwt.SignedJWT;
014    
015    import com.nimbusds.openid.connect.ParseException;
016    import com.nimbusds.openid.connect.SerializeException;
017    
018    import com.nimbusds.openid.connect.claims.ClientAuthenticationClaims;
019    import com.nimbusds.openid.connect.claims.ClientID;
020    
021    import com.nimbusds.openid.connect.http.CommonContentTypes;
022    import com.nimbusds.openid.connect.http.HTTPRequest;
023    
024    import com.nimbusds.openid.connect.util.URLUtils;

027    /**
028     * Base abstract class for JSON Web Token (JWT) based client authentication at 
029     * the Token endpoint.
030     *
031     * <p>See http://openid.net/specs/openid-connect-messages-1_0.html#client_authentication
032     *
033     * <p>See draft-ietf-oauth-assertions-03
034     *
035     * <p>See draft-jones-oauth-jwt-bearer-04
036     *
037     * @author Vladimir Dzhuvinov
038     * @version 0.9 (2012-05-16)
039     */
040    public abstract class JWTClientAuthentication extends ClientAuthentication {
041    
042    
043            /**
044             * The expected client assertion type, corresponding to the
045             * {@code client_assertion_type} parameter. This is a URN string set to
046             * "urn:ietf:params:oauth:client-assertion-type:jwt-bearer".
047             */
048            public static final String CLIENT_ASSERTION_TYPE = "urn:ietf:params:oauth:client-assertion-type:jwt-bearer";
049            
050    
051            /**
052             * The client assertion, corresponding to the {@link client_assertion}
053             * parameter. The assertion is in the form of a signed JWT.
054             */
055            private SignedJWT clientAssertion;
056            
057            
058            /**
059             * Optional client identifier, corresponding to the {@link client_id}
060             * parameter.
061             */
062            private ClientID clientID = null;
063            
064            
065            /**
066             * Creates a new JSON Web Token (JWT) based client authentication.
067             *
068             * @param method          The client authentication method. Must not be
069             *                        {@code null}.
070             * @param clientAssertion The client assertion, corresponding to the
071             *                        {@code client_assertion} parameter, in the 
072             *                        form of a signed JSON Web Token (JWT). Must 
073             *                        not be {@code null}.
074             * @param clientID        Optional client identifier, corresponding to
075             *                        the {@code client_id} parameter. {@code null}
076             *                        if not specified.
077             */
078            protected JWTClientAuthentication(final ClientAuthentication.Method method, 
079                                              final SignedJWT clientAssertion,
080                                              final ClientID clientID) {
081            
082                    super(ClientAuthentication.Method.CLIENT_SECRET_BASIC);
083            
084                    if (clientAssertion == null)
085                            throw new NullPointerException("The client assertion JWT must not be null");
086                            
087                    this.clientAssertion = clientAssertion;
088                    
089                    this.clientID = clientID;
090            }
091            
092            
093            /**
094             * Gets the client assertion, corresponding to the 
095             * {@code client_assertion} parameter.
096             *
097             * @return The client assertion, in the form of a signed JSON Web Token 
098             *         (JWT).
099             */
100            public SignedJWT getClientAssertion() {
101            
102                    return clientAssertion;
103            }
104            
105            
106            /**
107             * Gets the optional client identifier, corresponding to the
108             * {@code client_id} parameter.
109             *
110             * @return The client identifier, {@code null} if not specified.
111             */
112            public ClientID getClientID() {
113            
114                    return clientID;
115            }
116            
117            
118            /**
119             * Gets the client authentication claims contained in the client
120             * assertion JSON Web Token (JWT).
121             *
122             * @return The client authentication claims.
123             *
124             * @throws ParseException If the client assertion JSON Web Token (JWT)
125             *                        doesn't contain a valid client authentication
126             *                        claims set.
127             */
128            public ClientAuthenticationClaims getClientAuthenticationClaims()
129                    throws ParseException {
130            
131                    JSONObject claimsSet = clientAssertion.getClaimsSet().toJSONObject();
132                    
133                    if (claimsSet == null)
134                            throw new ParseException("Couldn't retrieve JSON object from the client assertion JWT");
135                    
136                    return ClientAuthenticationClaims.parse(claimsSet);
137            }
138            
139            
140            /**
141             * Returns the parameter representation of this JSON Web Token (JWT) 
142             * based client authentication. Note that the parameters are not 
143             * {@code application/x-www-form-urlencoded} encoded.
144             *
145             * <p>Parameters map:
146             *
147             * <pre>
148             * "client_assertion" -> [serialised-JWT]
149             * "client_assertion_type" -> "urn:ietf:params:oauth:client-assertion-type:jwt-bearer"
150             * "client_id" -> [optional-client-id]
151             * </pre>
152             *
153             * @return The parameters map, with keys "client_assertion",
154             *         "client_assertion_type" and "client_id".
155             *
156             * @throws SerializeException If the signed JWT couldn't be serialised
157             *                            to a client assertion string.
158             */
159            public Map<String,String> toParameters()
160                    throws SerializeException {
161            
162                    Map<String,String> params = new HashMap<String,String>();
163                    
164                    try {
165                            params.put("client_assertion", clientAssertion.serialize());
166                    
167                    } catch (JWTException e) {
168                    
169                            throw new SerializeException("Couldn't serialize JWT to a client assertion string: " + e.getMessage(), e);
170                    }       
171                    
172                    params.put("client_assertion_type", CLIENT_ASSERTION_TYPE);
173                    
174                    if (clientID != null)
175                            params.put("client_id", clientID.getClaimValue());
176                    
177                    return params;
178            }
179            
180            
181            /**
182             * @inheritDoc
183             */
184            public void apply(final HTTPRequest httpRequest)
185                    throws SerializeException {
186                    
187                    if (httpRequest.getMethod() != HTTPRequest.Method.POST)
188                            throw new SerializeException("The HTTP request method must be POST");
189                    
190                    ContentType ct = httpRequest.getContentType();
191                    
192                    if (ct == null)
193                            throw new SerializeException("Missing HTTP Content-Type header");
194                    
195                    if (! ct.match(CommonContentTypes.APPLICATION_URLENCODED))
196                            throw new SerializeException("The HTTP Content-Type header must be " + CommonContentTypes.APPLICATION_URLENCODED);
197                    
198                    Map <String,String> params = httpRequest.getQueryParameters();
199                    
200                    params.putAll(toParameters());
201                    
202                    String queryString = URLUtils.serializeParameters(params);
203                    
204                    httpRequest.setQuery(queryString);
205            }
206            
207            
208            /**
209             * Ensures the specified parameters map contains an entry with key 
210             * "client_assertion_type" pointing to a string that equals the expected
211             * {@link #CLIENT_ASSERTION_TYPE}. This method is intended to aid 
212             * parsing of JSON Web Token (JWT) based client authentication objects.
213             *
214             * @param params The parameters map to check. The parameters must not be
215             *               {@code null} and 
216             *               {@code application/x-www-form-urlencoded} encoded.
217             *
218             * @throws ParseException If expected "client_assertion_type" entry 
219             *                        wasn't found.
220             */
221            protected static void ensureClientAssertionType(final Map<String,String> params)
222                    throws ParseException {
223                    
224                    final String clientAssertionType = params.get("client_assertion_type");
225                    
226                    if (clientAssertionType == null)
227                            throw new ParseException("Missing \"client_assertion_type\" parameter");
228                    
229                    if (! clientAssertionType.equals(CLIENT_ASSERTION_TYPE))
230                            throw new ParseException("Invalid \"client_assertion_type\" parameter, must be " + CLIENT_ASSERTION_TYPE);
231            }
232            
233            
234            /**
235             * Parses the specified parameters map for a client assertion. This
236             * method is intended to aid parsing of JSON Web Token (JWT) based 
237             * client authentication objects.
238             *
239             * @param params The parameters map to parse. It must contain an entry
240             *               with key "client_assertion" pointing to a string that
241             *               represents a signed serialised JSON Web Token (JWT).
242             *               The parameters must not be {@code null} and
243             *               {@code application/x-www-form-urlencoded} encoded.
244             *
245             * @return The client assertion as a signed JSON Web Token (JWT).
246             *
247             * @throws ParseException If a "client_assertion" entry couldn't be
248             *                        retrieved from the parameters map.
249             */
250            protected static SignedJWT parseClientAssertion(final Map<String,String> params)
251                    throws ParseException {
252                    
253                    final String clientAssertion = params.get("client_assertion");
254                    
255                    if (clientAssertion == null)
256                            throw new ParseException("Missing \"client_assertion\" parameter");
257                    
258                    try {
259                            return SignedJWT.parse(clientAssertion);
260                            
261                    } catch (JWTException e) {
262                    
263                            throw new ParseException("Invalid \"client_assertion\" JWT: " + e.getMessage(), e);
264                    }
265            }
266            
267            /**
268             * Parses the specified parameters map for an optional client 
269             * identifier. This method is intended to aid parsing of JSON Web Token 
270             * (JWT) based client authentication objects.
271             *
272             * @param params The parameters map to parse. It may contain an entry
273             *               with key "client_id" pointing to a string that 
274             *               represents the client identifier. The parameters must 
275             *               not be {@code null} and 
276             *               {@code application/x-www-form-urlencoded} encoded.
277             *
278             * @return The client identifier, {@code null} if not specified.
279             */
280            protected static ClientID parseClientID(final Map<String,String> params) {
281                    
282                    final String clientIDString = params.get("client_id");
283                    
284                    if (clientIDString == null)
285                            return null;
286                    
287                    ClientID clientID = new ClientID();
288                    clientID.setClaimValue(clientIDString);
289                    return clientID;
290            }
291            
292            
293            /**
294             * Parses the specified HTTP request for a JSON Web Token (JWT) based
295             * client authentication.
296             *
297             * @param httpRequest The HTTP request to parse. Must not be {@code null}.
298             *
299             * @return The JSON Web Token (JWT) based client authentication.
300             *
301             * @throws ParseException If a JSON Web Token (JWT) based client 
302             *                        authentication couldn't be retrieved from the
303             *                        HTTP request.
304             */
305            public static JWTClientAuthentication parse(final HTTPRequest httpRequest)
306                    throws ParseException {
307                    
308                    httpRequest.ensureMethod(HTTPRequest.Method.POST);
309                    httpRequest.ensureContentType(CommonContentTypes.APPLICATION_URLENCODED);
310                    
311                    String query = httpRequest.getQuery();
312                    
313                    if (query == null)
314                            throw new ParseException("Missing HTTP POST request entity body");
315                    
316                    Map<String,String> params = URLUtils.parseParameters(query);
317                    
318                    JWA alg = parseClientAssertion(params).getHeader().getAlgorithm();
319                            
320                    if (ClientSecretJWT.getSupportedJWAs().contains(alg))
321                            return ClientSecretJWT.parse(params);
322                                    
323                    else if (PrivateKeyJWT.getSupportedJWAs().contains(alg))
324                            return PrivateKeyJWT.parse(params);
325                            
326                    else
327                            throw new ParseException("Unsupported signed JWT algorithm: " + alg);
328            }
