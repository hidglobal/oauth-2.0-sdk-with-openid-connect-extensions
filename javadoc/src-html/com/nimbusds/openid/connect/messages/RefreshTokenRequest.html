001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import java.util.LinkedHashMap;
005    import java.util.Map;
006    
007    import com.nimbusds.openid.connect.ParseException;
008    import com.nimbusds.openid.connect.SerializeException;
009    
010    import com.nimbusds.openid.connect.http.CommonContentTypes;
011    import com.nimbusds.openid.connect.http.HTTPRequest;
012    
013    import com.nimbusds.openid.connect.util.URLUtils;
014    
015    
016    /**
017     * Refresh token request to the Token endpoint.
018     *
019     * <p>Note that the optional scope parameter is not supported.
020     *
021     * <p>Example refresh token request:
022     *
023     * <pre>
024     * POST /token HTTP/1.1
025     * Host: server.example.com
026     * Authorization: Basic czZCaGRSa3F0MzpnWDFmQmF0M2JW
027     * Content-Type: application/x-www-form-urlencoded;charset=UTF-8
028     *
029     * grant_type=refresh_token&refresh_token=tGzv3JOkF0XG5Qx2TlKWIA
030     * </pre>
031     *
032     * <p>See draft-ietf-oauth-v2-26, section 6.
033     *
034     * @author Vladimir Dzhuvinov
035     * @version 0.9 (2012-05-11)
036     */
037    public class RefreshTokenRequest extends TokenRequest {
038    
039    
040            /**
041             * The refresh token.
042             */
043            private RefreshToken refreshToken;
044            
045            
046            /**
047             * Creates a new unauthenticated refresh token request.
048             *
049             * @param refreshToken The refresh token. Must not be {@code null}.
050             */
051            public RefreshTokenRequest(final RefreshToken refreshToken) {
052            
053                    this(refreshToken, null);
054            }
055            
056             
057            /**
058             * Creates a new authenticated refresh token request.
059             *
060             * @param refreshToken The refresh token. Must not be {@code null}.
061             * @param clientAuth   The client authentication, {@code null} if none.
062             */
063            public RefreshTokenRequest(final RefreshToken refreshToken, final ClientAuthentication clientAuth) {
064            
065                    super(GrantType.REFRESH_TOKEN, clientAuth);
066                    
067                    if (refreshToken == null)
068                            throw new NullPointerException("The refresh token must not be null");
069                    
070                    this.refreshToken = refreshToken;
071            }
072            
073            
074            /**
075             * Returns the HTTP request for this refresh token request.
076             *
077             * @return The HTTP request.
078             *
079             * @throws SerializeException If this refresh token request couldn't be 
080             *                            serialised to an HTTP request.
081             */
082            public HTTPRequest toHTTPRequest()
083                    throws SerializeException {
084                    
085                    HTTPRequest httpRequest = new HTTPRequest(HTTPRequest.Method.POST);
086                    
087                    httpRequest.setContentType(CommonContentTypes.APPLICATION_URLENCODED);
088                    
089                    Map<String,String> params = new LinkedHashMap<String,String>();
090                    params.put("grant_type", getGrantType().toString());
091                    params.put("refresh_token", refreshToken.toString());
092                    
093                    httpRequest.setQuery(URLUtils.serializeParameters(params));
094                    
095                    if (getClientAuthentication() != null)
096                            getClientAuthentication().apply(httpRequest);
097                    
098                    return httpRequest;
099            }
100            
101            
102            /**
103             * Parses the specified HTTP request for a refresh token request.
104             *
105             * @param httpRequest The HTTP request. Must not be {@code null}.
106             *
107             * @return The refresh token request.
108             *
109             * @throws ParseException If the HTTP request couldn't be parsed to a 
110             *                        valid refresh token request.
111             */
112            public static RefreshTokenRequest parse(final HTTPRequest httpRequest)
113                    throws ParseException {
114                    
115                    // Only HTTP POST accepted
116                    httpRequest.ensureMethod(HTTPRequest.Method.POST);
117                    httpRequest.ensureContentType(CommonContentTypes.APPLICATION_URLENCODED);
118                    
119                    // No fragment!
120                    // May use query component!
121                    
122                    String query = httpRequest.getQuery();
123                    
124                    if (query == null)
125                            throw new ParseException("Missing HTTP POST request entity body");
126                    
127                    Map<String,String> params = URLUtils.parseParameters(query);
128                    
129                    
130                    // Parse grant type
131                    final String grantTypeString = params.get("grant_type");
132                    
133                    if (grantTypeString == null)
134                            throw new ParseException("Missing \"grant_type\" parameter");
135                            
136                    if (! grantTypeString.equals(GrantType.AUTHORIZATION_CODE.toString()))
137                            throw new ParseException("Invalid \"grant_type\" parameter: " + grantTypeString);
138                    
139                    
140                    // Parse refresh token
141                    final String tokenString = params.get("refresh_token");
142                    
143                    if (tokenString == null)
144                            throw new ParseException("Missing \"refresh_token\" parameter");
145                    
146                    // Parse client authentication
147                    ClientAuthentication clientAuth = ClientAuthentication.parse(httpRequest);
148                    
149                    return new RefreshTokenRequest(new RefreshToken(tokenString), clientAuth);
150            }
151    }
