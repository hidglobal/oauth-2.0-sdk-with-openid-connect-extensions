001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import java.util.Collection;
005    import java.util.HashMap;
006    import java.util.Map;
007    
008    import com.nimbusds.openid.connect.ParseException;
009    
010    
011    /**
012     * Parser for {@link ScopeMember}s.
013     *
014     * @author Vladimir Dzhuvinov
015     * @version 0.9 (2012-04-22)
016     */
017    public class ScopeParser {
018    
019    
020            /**
021             * Parser instance for {@link StdScopeMember standard scope member}
022             * (thread-safe).
023             */
024            protected static final ScopeParser STD_SCOPE_PARSER = new ScopeParser();
025            
026    
027            /**
028             * Map of scope value strings to their corresponding object.
029             */
030            private Map<String,ScopeMember> map = new HashMap<String,ScopeMember>();
031            
032            
033            /**
034             * Creates a new parser for the {@link StdScopeMember standard scope
035             * members}.
036             */
037            public ScopeParser() {
038            
039                    this(StdScopeMember.values());
040            }
041            
042            
043            /**
044             * Creates a new parser for the specified scope members.
045             *
046             * @param members The scope members to parse.
047             */
048            public ScopeParser(final ScopeMember... members) {
049            
050                    for (ScopeMember m: members)
051                            map.put(m.toString(), m);
052            }
053            
054            
055            /**
056             * Returns the scope members configured for parsing.
057             *
058             * @return The scope members for parsing.
059             */
060            public Collection<ScopeMember> getValues() {
061            
062                    return map.values();
063            }
064            
065            
066            /**
067             * Parses a {@code Scope} from the specified string. Unexpected scope
068             * members are ignored. For strict parsing see {@link #parseStrict}.
069             *
070             * @param s A string containing one or more scope members delimited by 
071             *          space.
072             *
073             * @return The parsed scope.
074             */
075            public Scope parse(final String s) {
076            
077                    Scope scope = new Scope();
078            
079                    String[] tokens = s.split("\\s+");
080                    
081                    for (String t: tokens) {
082                    
083                            if (map.containsKey(t))
084                                    scope.add(map.get(t));
085                    }
086                    
087                    return scope;
088            }
089            
090            
091            /**
092             * Parses a {@code Scope} from the specified string.
093             *
094             * @param s A string containing one or more scope members delimited by 
095             *          space.
096             *
097             * @return The parsed scope.
098             *
099             * @throws ParseException If an unexpected scope member is encountered
100             *                        or a {@link StdScopeMember#OPENID} is missing.
101             */
102            public Scope parseStrict(final String s)
103                    throws ParseException {
104                    
105                    Scope scope = new Scope();
106                    
107                    String[] tokens = s.split("\\s+");
108                    
109                    for (String t: tokens) {
110                    
111                            if (map.containsKey(t))
112                                    scope.add(map.get(t));
113                            else
114                                    throw new ParseException("Couldn't parse scope: Unexpected scope member: " + t);
115                    }
116                    
117                    if (! scope.isValid())
118                            throw new ParseException("Couldn't parse scope: Missing \"openid\" scope member");
119                    
120                    return scope;
121            }
122    }
