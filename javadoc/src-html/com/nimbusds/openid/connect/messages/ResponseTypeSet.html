001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import java.util.HashSet;
005    import java.util.Set;
006    
007    import com.nimbusds.openid.connect.ParseException;
008    
009    
010    /**
011     * Set of {@link ResponseType}s.
012     *
013     * @author Vladimir Dzhuvinov
014     * @version 0.9 (2012-04-22)
015     */
016    public class ResponseTypeSet extends HashSet<ResponseType> {
017    
018    
019            /**
020             * Creates a new empty response type set.
021             */
022            public ResponseTypeSet() {
023            
024                    // Nothing to do
025            }
026            
027            
028            /**
029             * Parses a set of authorisation response types.
030             *
031             * <p>OpenID Connect specifies the following string vectors but this 
032             * method will allow other orders as well to accommodate sloppy 
033             * serialisers:
034             *
035             * <pre>
036             * code
037             * token
038             * id_token
039             * id_token token
040             * code token
041             * code id_token token
042             * </pre>
043             *
044             * @param s Space-delimited list of one or more individual authorisation
045             *          response types.
046             *
047             * @return The parsed authorisation response types set.
048             *
049             * @throws ParseException If the parsed string is {@code null}, empty or
050             *                        contains an invalid response type name.
051             */
052            public static ResponseTypeSet parse(final String s)
053                    throws ParseException {
054            
055                    if (s == null || s.trim().isEmpty())
056                            throw new ParseException("Couldn't parse response type: Null or empty string");
057            
058                    ResponseTypeSet set = new ResponseTypeSet();
059                    
060                    String[] tokens = s.split("\\s+");
061                    
062                    for (String t: tokens)
063                            set.add(ResponseType.parse(t));
064                    
065                    return set;
066            }
067            
068            
069            /**
070             * Returns the canonical string representation of this set of 
071             * authorisation response types.
072             *
073             * <p>The serialised set is guaranteed to have one of the following
074             * orders (to conform with the OpenID Connect spec):
075             *
076             * <pre>
077             * code
078             * token
079             * id_token
080             * id_token token
081             * code token
082             * code id_token token
083             * </pre>
084             *
085             * @return Space-delimited string representing the authorisation 
086             *         response types.
087             */
088            public String toString() {
089            
090                    StringBuilder sb = new StringBuilder();
091                    
092                    if (contains(ResponseType.CODE)) {
093                    
094                            sb.append("code");
095                    }
096                    
097                    if (contains(ResponseType.ID_TOKEN)) {
098                    
099                            if (sb.length() > 0)
100                                    sb.append(" id_token");
101                            else
102                                    sb.append("id_token");
103                    }
104                    
105                    if (contains(ResponseType.TOKEN)) {
106                    
107                            if (sb.length() > 0)
108                                    sb.append(" token");
109                            else
110                                    sb.append("token");
111                    }
112            
113                    return sb.toString();
114            }
115    }
