832            /**
833             * Puts the speicifed string claims with optional language tags into a
834             * JSON object.
835             *
836             * <p>Example:
837             * 
838             * <pre>
839             * {
840             *   "country"       : "USA",
841             *   "country#en"    : "USA",
842             *   "country#de_DE" : "Vereinigte Staaten",
843             *   "country#fr_FR" : "Etats Unis"
844             * }
845             * </pre>
846             *
847             * @param o      The JSON object. May be {@code null}.
848             * @param claims The claims. May be {@code null}.
849             */
850            protected static void putIntoJSONObject(final JSONObject o, final Map<LangTag,? extends StringClaimWithLangTag> claims) {
851            
852                    if (o == null || claims == null)
853                            return;
854                    
855                    Iterator <? extends StringClaimWithLangTag> it = claims.values().iterator();
856                    
857                    while (it.hasNext()) {
858                    
859                            StringClaimWithLangTag claim = it.next();
860                            
861                            o.put(claim.getClaimName(), claim.getClaimValue().toString());
862                    }
863            }

870
