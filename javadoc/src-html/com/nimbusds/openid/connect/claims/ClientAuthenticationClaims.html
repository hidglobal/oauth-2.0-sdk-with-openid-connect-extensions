017    /**
018     * Client authentication claims, serialisable to a JSON object. Used for 
019     * {@link com.nimbusds.openid.connect.messages.ClientSecretJWT client secret 
020     * JWT} and {@link com.nimbusds.openid.connect.messages.PrivateKeyJWT private 
021     * key JWT} authentication at the Token endpoint.
022     *
023     * <p>Note that OpenID Connect mandates the use the {@code jti} claim, see
024     * https://bitbucket.org/openid/connect/issue/583/messages-221-client-auth-claims-not
025     *
026     * <p>Example client authentication claims set:
027     *
028     * <pre>
029     * {
030     *   "iss" : "http://client.example.com",
031     *   "prn" : "http://client.example.com",
032     *   "aud" : "http://server.example.com",
033     *   "jti" : "d396036d-c4d9-40d8-8e98-f7e8327002d9",
034     *   "exp" : 1311281970,
035     *   "iat" : 1311280970
036     * }
037     * </pre>
038     *
039     * <p>See http://openid.net/specs/openid-connect-messages-1_0.html#client_authentication
040     *
041     * <p>See draft-jones-oauth-jwt-bearer-04
042     *
043     * @author Vladimir Dzhuvinov
044     * @version 0.9 (2012-05-11)
045     */
046    public class ClientAuthenticationClaims extends JSONObjectClaims {

057                    reservedClaimNames.add("prn");
058                    reservedClaimNames.add("aud");
059                    reservedClaimNames.add("jti");
060                    reservedClaimNames.add("exp");
061                    reservedClaimNames.add("iat");
062                    reservedClaimNames.add("nbf");
063            }

073            
074                    return Collections.unmodifiableSet(reservedClaimNames);

133             *            OAuth client. Must not be {@code null}.
134             * @param prn The principal. Must contain the {@code client_id} of the

136             * @param aud The audience, typically the URL of the authorisation

139             *            by implementations requiring message de-duplication for 
140             *            one-time use assertions.
141             * @param exp The expiration time. Must not be {@code null}.
142             */
143            public ClientAuthenticationClaims(final Issuer iss, 
144                                              final Principal prn, 
145                                              final Audience aud,
146                                              final JWTID jti,
147                                              final ExpirationTime exp) {
148    
149                    setIssuer(iss);
150                    setPrincipal(prn);
151                    setAudience(aud);
152                    setJWTID(jti);
153                    setExpirationTime(exp);
154            }

340            /**
341             * @inheritDoc
342             */
343            public void addCustomClaim(final GenericClaim customClaim) {
344            
345                    if (reservedClaimNames.contains(customClaim.getClaimName()))
346                            throw new IllegalArgumentException("Custom claim name conflicts with reserved claim name: " + customClaim.getClaimName());
347            
348                    customClaims.put(customClaim.getClaimName(), customClaim);
349            }

352            /**
353             * @inheritDoc
354             */
355            public JSONObject toJSONObject() {
356            
357                    JSONObject o = new JSONObject();
358                    
359                    o.put("iss", iss.getClaimValue());
360                    o.put("prn", prn.getClaimValue());
361                    o.put("aud", aud.getClaimValue());
362                    o.put("jti", jti.getClaimValue());
363                    o.put("exp", exp.getClaimValue());
364                    
365                    if (iat != null)
366                            o.put("iat", iat.getClaimValue());
367                    
368                    if (nbf != null)
369                            o.put("nbf", nbf.getClaimValue());
370                    
371                    return o;
372            }

375            /**
376             * Parses a client authentication claims set from the specified JSON 
377             * object.
378             *
379             * @param jsonObject The JSON object to parse. Must not be {@code null}.
380             *
381             * @return The client authentication claims set.
382             *
383             * @throws ParseException If the JSON object cannot be parsed to a valid
384             *                        client authentication claims set.
385             */
386            public static ClientAuthenticationClaims parse(final JSONObject jsonObject)
387                    throws ParseException {
388                    
389                    // Get required ID token claims
390                    
391                    Issuer iss = new Issuer();
392                    ClaimValueParser.parse(jsonObject, iss);
393                    jsonObject.remove(iss.getClaimName());
394                    
395                    Principal prn = new Principal();
396                    ClaimValueParser.parse(jsonObject, prn);
397                    jsonObject.remove(prn.getClaimName());
398                    
399                    Audience aud = new Audience();
400                    ClaimValueParser.parse(jsonObject, aud);
401                    jsonObject.remove(aud.getClaimName());
402                    
403                    ExpirationTime exp = new ExpirationTime();
404                    ClaimValueParser.parse(jsonObject, exp);
405                    jsonObject.remove(exp.getClaimName());
406                    
407                    JWTID jti = new JWTID();
408                    ClaimValueParser.parse(jsonObject, jti);
409                    jsonObject.remove(jti.getClaimName());
410                    
411                    
412                    ClientAuthenticationClaims clientAuthClaims = 
413                            new ClientAuthenticationClaims(iss, prn, aud, jti, exp);
414                    
415                    
416                    // Get optional iat claim
417                    IssueTime iat = new IssueTime();
418                    
419                    if (jsonObject.containsKey(iat.getClaimName())) {
420                            
421                            ClaimValueParser.parse(jsonObject, iat);
422                            jsonObject.remove(iat.getClaimName());
423                            clientAuthClaims.setIssueTime(iat);
424                    }
425                    
426                    
427                    // Get optional nbf claim
428                    NotBeforeTime nbf = new NotBeforeTime();
429                    
430                    if (jsonObject.containsKey(nbf.getClaimName())) {
431                    
432                            ClaimValueParser.parse(jsonObject, nbf);
433                            jsonObject.remove(nbf.getClaimName());
434                            clientAuthClaims.setNotBeforeTime(nbf);
435                    }
436                    
437                    
438                    // Add remaing claims as custom
439                    
440                    Iterator <Map.Entry<String,Object>> it = jsonObject.entrySet().iterator();
441                    
442                    while (it.hasNext()) {
443                    
444                            Map.Entry <String,Object> entry = it.next();
445                            
446                            GenericClaim gc = new GenericClaim(entry.getKey());
447                            gc.setClaimValue(entry.getValue());
448                            
449                            clientAuthClaims.addCustomClaim(gc);
450                    }
451                    
452                    return clientAuthClaims;
453            }
